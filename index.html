update() { // Nếu đã cố định thì không di chuyển nữa if (this.fixed) { this.draw(); return; } // Di chuyển ngẫu nhiên this.x += this.vx; this.y += this.vy; if (this.x <= 0 || this.x >= width) this.vx *= -1; if (this.y <= 0 || this.y >= height) this.vy *= -1; // Khi con trỏ ở gần, hạt trở về vị trí gốc if (mouse.x !== null && mouse.y !== null) { const dxMouse = mouse.x - this.x; const dyMouse = mouse.y - this.y; const distance = Math.sqrt(dxMouse * dxMouse + dyMouse * dyMouse); if (distance < mouse.radius) { const dx = this.baseX - this.x; const dy = this.baseY - this.y; this.x += dx * CONFIG.returnSpeed; this.y += dy * CONFIG.returnSpeed; // Khi đã về đúng vị trí, khóa lại if (Math.abs(dx) < 0.5 && Math.abs(dy) < 0.5) { this.x = this.baseX; this.y = this.baseY; this.vx = 0; this.vy = 0; this.fixed = true; // Khóa vĩnh viễn } } } this.draw(); }
